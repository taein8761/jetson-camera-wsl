cmake_minimum_required(VERSION 3.16)
project(camera2_3)

# Default to C99
if(NOT CMAKE_C_STANDARD)
  set(CMAKE_C_STANDARD 99)
endif()

# Default to C++14
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 14)
endif()

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(cv_bridge REQUIRED)
find_package(OpenCV REQUIRED)

#build
add_executable(pmain src/pmain.cpp src/pub.cpp)
ament_target_dependencies(pmain rclcpp sensor_msgs cv_bridge OpenCV)

add_executable(smain src/smain.cpp src/sub.cpp)
ament_target_dependencies(smain rclcpp sensor_msgs OpenCV)

# Install
install(TARGETS
  pmain
  smain
  DESTINATION lib/${PROJECT_NAME})

ament_package()
